    example
    POST /test HTTP/1.1
    Host: whatap.requestcatcher.com
    Accept: text/plain, application/json, application/*+json, */*
    Connection: keep-alive
    Content-Length: 234
    Content-Type: application/json
    User-Agent: Java/17.0.7
    User-Header-Key: Test-Value
